% Colonna centrata espandibile
\newcolumntype{Y}{>{\centering\arraybackslash}X}

% Contatore per le righe
\newcounter{dndrowcount}

% Salva comandi originali
\let\OriTabularx\tabularx
\let\endOriTabularx\endtabularx
\let\OriNewline\\

% Comando per gestire colori automaticamente
\newcommand{\dndautorow}{%
	\ifnum\value{dndrowcount}=0%
	\rowcolor{dndheader}% Header (riga 0)
	\else%
	\ifodd\value{dndrowcount}%
	\rowcolor{white}% Righe dispari
	\else%
	\rowcolor{dndalt}% Righe pari
	\fi%
	\fi%
}

% Ridefinizione di tabularx
\renewenvironment{tabularx}[2]{%
	\setcounter{dndrowcount}{0}% Reset contatore
	\par\vspace{0.5em}\noindent%
	\renewcommand{\arraystretch}{1.2}%
	\setlength{\tabcolsep}{6pt}%
	% Ridefinisci \\ solo per questa tabella
	\def\\{%
		\OriNewline\hline%
		\stepcounter{dndrowcount}%
		\dndautorow%
	}%
	\OriTabularx{#1}{|#2|}%
	\hline%
	\dndautorow% Colora la prima riga (header)
}{%
	% Nessuna riga finale - finisce pulito
	\endOriTabularx%
	\vspace{0.5em}%
	\let\\\OriNewline% Ripristina \\ originale
}
